\begin{verbatim}
df["YoY_GROWTH"] = np.log(df["REAL_GDP"]) \
                 - np.log(df["REAL_GDP"]).shift(4)
\end{verbatim}

\item Produce forecasts for 2022Q3 and 2022Q4, both manually and using Python. Do this for
\begin{enumerate}
	\item an MA(3) model, and
	\item an AR(1) model.
